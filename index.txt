<?php
if(isset($_POST['submit'])){
    $a = $_POST['a'];
    $b = $_POST['b'];
    $c = $_POST['c'];
    $discriminant = pow($b, 2) - 4 * $a * $c;

    echo '<div class="container">';
    echo '<h1>Quadratic Discriminant Result</h1>';
    echo '<p>The discriminant is: ' . $discriminant . '</p>';
